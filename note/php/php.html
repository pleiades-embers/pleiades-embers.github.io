    # PHP运行出现Notice : Notice: Undefined index的原因及解决办法

    搜索php.ini：

    error_reporting = E_ALL

    改为：

    error_reporting = E_ALL & ~E_NOTICE

    还有个不是办法的办法就是

    在每个文件头上加

    error_reporting(0); 虽然不好弄但是可以解决问题

    
<?php
error_reporting(0);
$u=$_GET['u'];
echo $u;
echo "<br>";
//加上error_reporting(0);就不会弹出警告了
?>
